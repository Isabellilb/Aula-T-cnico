import kotlin.random.Random

fun celsiusParaFahrenheit(celsius: Int): Double {
    return (celsius * 9 / 5.0) + 32
}

fun main() {
    val celsius = Random.nextInt(-10, 41) // Gera temperatura entre -10째C e 40째C
    val fahrenheit = celsiusParaFahrenheit(celsius)
    println("Temperatura: $celsius째C = %.2f째F".format(fahrenheit))
}
